S'observa que els preus s'han incrementat des dels 2,20 euros de 2005 fins al 4,60 en 2017 (darrer any en que aquesta marca va variar el preu). En 2006 i 2011 va haver davallades degut a una "guerra de preus" en el mercat de tabac que l'Estat va tallar augmentant els impostos sobre les labors de tabac.

# Analitzem l'evoluci de les vendes en unitats i preu
df_vendes = pd.read_csv('TabacoVentas.csv', sep=';')

def serie_vendes(df_vendes, labor, unitat):
    df_series = df_vendes[(df_vendes.Unidad==unitat) & (df_vendes.Labor==labor)]
    df_series = df_series.groupby('Anyo').sum()
    return df_series.reset_index('Anyo').reindex(columns=['Anyo', 'Total'])

df_vendes_cig = serie_vendes(df_vendes, 'CIGARRILLOS', 'cantidad')
